2.2.3 实体类相关代码

SearchServiceImpl 相关代码

package com.pinyougou.search.service.impl;

import com.alibaba.dubbo.config.annotation.Service;
import com.pinyougou.pojo.TbItem;
import com.pinyougou.search.service.SearchService;
import org.apache.commons.lang3.StringUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Sort;
import org.springframework.data.redis.core.RedisTemplate;
import org.springframework.data.solr.core.SolrTemplate;
import org.springframework.data.solr.core.query.*;
import org.springframework.data.solr.core.query.result.*;
import org.springframework.transaction.annotation.Transactional;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

@Service
@Transactional
public class SearchServiceImpl implements SearchService {

    @Autowired
    private SolrTemplate solrTemplate;
    @Autowired
    private RedisTemplate redisTemplate;

    @Override
    public Map search(Map searchEntity) {
        // 高亮显示需要用到的 SimpleHighlightQuery
        SimpleHighlightQuery query = new SimpleHighlightQuery();
        // 根据关键字查询
        Criteria criteria = new Criteria("item_keywords").is(searchEntity.get("keywords"));
        query.addCriteria(criteria);

        // 使用分组查询获取商品的分类并展示
        // 创建一个对象 ，其中需要放入 分组查询的条件
        SimpleQuery groupQuery = new SimpleQuery();
        // 设置查询条件 和高亮条件一致， 保证查询结果最终是一致的
        groupQuery.addCriteria(criteria);
        // 赋值分组查询的参数
        // 创建分组条件
        GroupOptions groupOptions = new GroupOptions();
        // 设置分组域名
        groupOptions.addGroupByField("item_category");
        groupQuery.setGroupOptions(groupOptions);
        GroupPage<TbItem> groupPage = solrTemplate.queryForGroupPage(groupQuery, TbItem.class);
        // 获取分组域名的结果
        GroupResult<TbItem> groupResult = groupPage.getGroupResult("item_category");
        // 获取数据格式的数组
        Page<GroupEntry<TbItem>> groupEntries = groupResult.getGroupEntries();
        List<GroupEntry<TbItem>> content = groupEntries.getContent();
        // 创建一个集合。讲循环获取的值放入到集合中
        ArrayList<String> categoryList = new ArrayList<>();
        // 循环多个对象 ，获取其中的内容
        for (GroupEntry<TbItem> groupEntry : content) {
            // 获取到其中的每一个值，创建一个集合，将内容放入其中，并且返会内容
            categoryList.add(groupEntry.getGroupValue());
        }


        // 根据商品的第一个分类查询改分类下的品牌和规格数据
        // 定义品牌集合
        List<Map> brandList = new ArrayList<>();
        // 定义规格集合
        List<Map> specList = new ArrayList<>();
        // 判断当前的长度是否大于 0
        if (categoryList.size() > 0) {
            // 获取到上面循环中品牌的名字
            String categoryName = categoryList.get(0);
            // 根据名字获取品牌，放入结果中
            brandList = (List<Map>) redisTemplate.boundHashOps("brandList").get(categoryName);
            // 根据名获取规格，放入结果中
            specList = (List<Map>) redisTemplate.boundHashOps("specList").get(categoryName);
        }







        //给返回的关键字进行高亮显示    setHighlightOptions  ：高亮显示的方法
        HighlightOptions highlightOptions = new HighlightOptions();
        // 设置需要高亮显示的域名
        highlightOptions.addField("item_title");
        // 设置高亮显示的前缀和后缀
        //标签样式前缀
        highlightOptions.setSimplePrefix("<font style='color:red'>");
        // 标签样式后缀
        highlightOptions.setSimplePostfix("</font>");

        // 需要一个  highlightOptions ， 所以创建 一个  highlightOptions
        query.setHighlightOptions(highlightOptions);


        // 添加过滤查询 category
        SimpleFilterQuery filterQuery = new SimpleFilterQuery();
        // 获取过滤条件判断是否存在
        String category = (String) searchEntity.get("category");
        if (StringUtils.isNotBlank(category)) {
            // 存在是，添加过滤条件中
            // 调用 addCriteria  方法添加过滤条件  里面需要一个参数  Criteria  条件
            // 如果有值 ，就表示  item_category 字段 = category   如果没有值，就表示  item_category = null
            filterQuery.addCriteria(new Criteria("item_category").is(category));
        }


        //获取过滤条件，判断是否存在  品牌
        String brand = (String) searchEntity.get("brand");
        if(org.apache.commons.lang.StringUtils.isNotBlank(brand)){
            //存在时，添加到过滤查询条件中
            filterQuery.addCriteria(new Criteria("item_brand").is(brand));
        }


        //获取过滤条件，判断是否存在  规格 和 规格附加属性
        // 从前台传过来的 searchEntity 中 获取 spec 的内容
        Map<String, String> specMap = (Map) searchEntity.get("spec");
        // 判断 map 集合是否为空
        if(null != specMap && specMap.keySet().size() > 0){
            // 遍历 map 集合
            for(String key : specMap.keySet()){
                // 通过 key 获取到 value 值
                String value = specMap.get(key);
                // 判断 value 不为空
                if(org.apache.commons.lang.StringUtils.isNotBlank(value)){
                    //存在时，添加到过滤查询条件中  item_spec_网络 is （3G，4G，2G）
                    filterQuery.addCriteria(new Criteria("item_spec_"+key).is(value));
                }
            }
        }


        // 过滤价格条件
        //获取过滤条件，判断是否存在
        String price = (String) searchEntity.get("price");
        if(org.apache.commons.lang.StringUtils.isNotBlank(price)){
            String[] priceArr = price.split("-");
            if("*".equals(priceArr[1])){
                //3000元以上
                //存在时，添加到过滤查询条件中
                filterQuery.addCriteria(new Criteria("item_price").greaterThanEqual(priceArr[0]));
            } else {
                //3000元以下的区间
                filterQuery.addCriteria(new Criteria("item_price").between(priceArr[0], priceArr[1]));
            }
        }


        // 添加条件  价格进行排序
        // 获取到前台传输过来的两个数据
        // 前台传输过来的排序状态
        String sort = (String) searchEntity.get("sort");
        // 需要进行排序的域
        String sortFiled = (String) searchEntity.get("sortFiled");
        // 判断当前两个值的状态是否为空
        if (StringUtils.isNotBlank(sort) && StringUtils.isNotBlank(sortFiled)) {
            // 不为空的话 ，就添加条件到 状态中
            // 判断当前 sort 的状态是 升序 还是 降序
            if ("ASC".equals(sort)) {
                // 升序排序   addSort()   :   方法       需要一个 sort   （两个参数）   ： 第一个是 升序还是降序   ， 第二个是要排序的主题
                query.addSort(new Sort(Sort.Direction.ASC,sortFiled));
            } else {
                // 降序排序
                query.addSort(new Sort(Sort.Direction.DESC,sortFiled));
            }
        }

        query.addFilterQuery(filterQuery);







        // 分页显示的方法
        // 获取到前台传过来的两个参数  page size
        Integer page = (Integer) searchEntity.get("page");
        Integer size = (Integer) searchEntity.get("size");
        // 判断两个数是否为空  如果为空的话  就手动进行设置默认值
        if (page != null && page <= 0) {
            // 数据不合法  当前页默认值设置为 1
            page = 1;
        }
        if (size != null && size <= 0) {
            // 数据不合法  每页显示条数默认为 8
            size = 8;
        }
        // 将分页的第一页添加到条件中    当前页的索引值
        //当前页第一条数据的索引值：（当前页-1）*每页条数
        query.setOffset((page - 1) * size);
        // 将分页的每页显示条数添加到条件中     每页条数
        query.setRows(size);





        /*
        *   queryForHighlightPage  高亮显示需要用到的 方法
        *       返回值对象时 一个原来的集合 和 一个 高亮集合对象
        * */
        HighlightPage<TbItem> tbItems = solrTemplate.queryForHighlightPage(query, TbItem.class);
        // 获取到所有的 高亮集合对象
        List<HighlightEntry<TbItem>> highlighted = tbItems.getHighlighted();
        // 进行遍历
        for (HighlightEntry<TbItem> tbItemHighlightEntry : highlighted) {
            // 获取到原对象 中的 title 属性 进行替换
            TbItem entity = tbItemHighlightEntry.getEntity();
            List<HighlightEntry.Highlight> highlights = tbItemHighlightEntry.getHighlights();
            // 判断 高亮集合对象下的  Highlights 对象是否为空 并且判断长度是否大于 0
            // 并且判断当前对象下的第一个下标值对象中的Snipplets 是否为空 并且 长度大于0
            if (highlights != null && highlights.size() > 0
                    && highlights.get(0).getSnipplets() != null && highlights.get(0).getSnipplets().size()>0) {
                // 获取当前对象下的第一个下标值中的第一个对象
                // 将获取的值 赋值到 entity中的 title 中
                entity.setTitle(highlights.get(0).getSnipplets().get(0));
            }

        }


        Map map = new HashMap();
        //总条数
        map.put("total", tbItems.getTotalElements());
        //每页显示内容的集合
        map.put("itemList", tbItems.getContent());
        //分类名的集合
        map.put("categoryList",categoryList);
        //品牌集合
        map.put("brandList",brandList);
        // 规格集合
        map.put("specList",specList);
        // 总页数
        map.put("totalPages", tbItems.getTotalPages());
        return map;
    }

}
